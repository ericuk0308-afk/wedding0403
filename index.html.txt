        // --- Book Page Turn Logic ---
        let currentPage = 0;
        const cardFront = document.querySelector('.card-front');
        const dot0 = document.getElementById('dot-0');
        const dot1 = document.getElementById('dot-1');

        window.flipToPage = (index) => {
            if (currentPage === index) return; 
            
            currentPage = index;
            
            // 重置動畫狀態
            cardFront.classList.remove('turn-forward', 'turn-backward');
            void cardFront.offsetWidth; 

            if (currentPage === 1) {
                cardFront.classList.add('turn-forward');
                dot0.classList.remove('active');
                dot1.classList.add('active');
            } else {
                cardFront.classList.add('turn-backward');
                dot1.classList.remove('active');
                dot0.classList.add('active');
            }
        };

        // Touch Swipe Handling for Flip
        let touchStartX = 0;
        let touchEndX = 0;
        const flipScene = document.getElementById('flip-scene');

        flipScene.addEventListener('touchstart', e => {
            touchStartX = e.changedTouches[0].screenX;
        }, {passive: true});

        flipScene.addEventListener('touchend', e => {
            touchEndX = e.changedTouches[0].screenX;
            handleSwipe();
        }, {passive: true});

        function handleSwipe() {
            const swipeThreshold = 50; 
            if (touchStartX - touchEndX > swipeThreshold) {
                flipToPage(1);
            } else if (touchEndX - touchStartX > swipeThreshold) {
                flipToPage(0);
            }
        }
